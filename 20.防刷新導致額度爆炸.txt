  function readField(field) {
    if (!field) return "";
    if ("stringValue" in field) return field.stringValue;
    if ("integerValue" in field) return Number(field.integerValue);
    if ("doubleValue" in field) return Number(field.doubleValue);
    if ("booleanValue" in field) return field.booleanValue;
    return "";
  }

  function writeFields(data) {
    const fields = {};
    Object.keys(data).forEach(key => {
      const value = data[key];
      if (typeof value === "boolean") fields[key] = { booleanValue: value };
      else if (typeof value === "number") fields[key] = { integerValue: String(value) };
      else fields[key] = { stringValue: String(value == null ? "" : value) };
    });
    return { fields };
  }

  async function apiFetch(url, options) {
    options = options || {};
    options.headers = Object.assign({ "Content-Type": "application/json" }, options.headers || {});
    if (app.idToken && !options.skipAuthHeader) options.headers.Authorization = "Bearer " + app.idToken;

    if (!response.ok) throw new Error((data.error && data.error.message) || response.statusText);
    return data;
  }

      }
    }

    const url = "https://identitytoolkit.googleapis.com/v1/accounts:signUp?key=" + firebaseConfig.apiKey;
    const data = await apiFetch(url, { method: "POST", skipAuthHeader: true, body: JSON.stringify({ returnSecureToken: true }) });
    app.idToken = data.idToken;
    app.refreshToken = data.refreshToken;
    app.user = null;
    saveAnonymousAuth(data);
  }

  async function getAccountInfo(idToken) {
    const url = "https://identitytoolkit.googleapis.com/v1/accounts:lookup?key=" + firebaseConfig.apiKey;
    const data = await apiFetch(url, {
      method: "POST",
      skipAuthHeader: true,
      body: JSON.stringify({ idToken })
    });
    return data.users && data.users[0] ? data.users[0] : null;
  }

  function hasAdminClaim(userInfo) {
    if (!userInfo || !userInfo.customAttributes) return false;
    try {
      return JSON.parse(userInfo.customAttributes).admin === true;
    } catch (_) {
      return false;
    }
  }

  async function signInWithAdmin(email, password) {
    const url = "https://identitytoolkit.googleapis.com/v1/accounts:signInWithPassword?key=" + firebaseConfig.apiKey;
    const data = await apiFetch(url, {
      method: "POST",
      skipAuthHeader: true,
      body: JSON.stringify({ email, password, returnSecureToken: true })
    });
    const userInfo = await getAccountInfo(data.idToken);
    if (!hasAdminClaim(userInfo)) {
      await signInAnonymous();
      throw new Error("此帳號尚未被設定為管理員。請先用 Node 建立或授權 admin 帳號。");
    }
    app.idToken = data.idToken;
    app.refreshToken = data.refreshToken;
    app.user = { email: data.email, admin: true };
  }

  async function loadStudents() {
    if (app.user && app.user.admin) {
      // 管理員模式：拉取完整資料庫 (消耗 N 次讀取)
      const data = await apiFetch(firestoreBase() + "/students?pageSize=500");
      app.students = (data.documents || []).map(docToStudent).filter(s => s.id !== "__directory__");
    } else {
      // 公開模式：只拉取一份總表 (只消耗 1 次讀取！)
      try {
        const data = await apiFetch(firestoreBase() + "/students/__directory__");
        if (data.fields && data.fields.data) {
          app.students = JSON.parse(data.fields.data.stringValue);
        } else {
          app.students = [];
        }
      } catch (error) {
        if (error.message && error.message.includes("404")) app.students = [];
        else throw error;
      }
    }
    render();
  }

  async function createStudent(student) {
    await apiFetch(firestoreBase() + "/students", { method: "POST", body: JSON.stringify(writeFields(student)) });
    await loadStudents();
  }

  async function patchStudent(id, fields) {
    const mask = Object.keys(fields).map(k => "updateMask.fieldPaths=" + encodeURIComponent(k)).join("&");
    const updatedDoc = await apiFetch(firestoreBase() + "/students/" + id + "?" + mask, {
      method: "PATCH",
      body: JSON.stringify(writeFields(fields))
    });
    
    // 本地更新資料，避免重新讀取整個資料庫導致額度爆表
    const index = app.students.findIndex(s => s.id === id);
    if (index !== -1 && updatedDoc.name) {
      app.students[index] = docToStudent(updatedDoc);
      render();
    }
  }

  async function deleteStudent(id) {
    await apiFetch(firestoreBase() + "/students/" + id, { method: "DELETE" });
    await loadStudents();
  }

  async function commitWrites(writes) {
    await apiFetch(firestoreBase().replace("/documents", "/documents:commit"), {
      method: "POST",
      body: JSON.stringify({ writes })
    });
    await loadStudents();
  }

  function documentName(id) {
    return "projects/" + firebaseConfig.projectId + "/databases/(default)/documents/students/" + id;
  }

  async function incrementStudent(id, key) {
    const student = app.students.find(s => s.id === id);
    const next = Number((student && student[key]) || 0) + 1;
    await patchStudent(id, { [key]: next });
  }

  function checkRefreshLock() {
    const now = Date.now();
    const lockUntil = parseInt(localStorage.getItem("app_lock_until") || "0", 10);
    if (lockUntil > now) {
      app.locked = true;
      app.lockSeconds = Math.ceil((lockUntil - now) / 1000);
      startLockTimer();
      return true;
    }
    if (lockUntil > 0) localStorage.removeItem("app_lock_until");

    let timestamps = readJson("app_refresh_timestamps", []);
    timestamps = timestamps.filter(t => now - t < 30000);
    timestamps.push(now);
    if (timestamps.length > 3) {
      const until = now + 5 * 60 * 1000;
      localStorage.setItem("app_lock_until", String(until));
      app.locked = true;
      app.lockSeconds = 300;
      startLockTimer();
      return true;
    }
    saveJson("app_refresh_timestamps", timestamps);
    return false;
  }

  function startLockTimer() {
    window.clearInterval(app.lockTimer);
    app.lockTimer = window.setInterval(() => {
      app.lockSeconds -= 1;
      if (app.lockSeconds <= 0) {
        localStorage.removeItem("app_lock_until");
        localStorage.removeItem("app_refresh_timestamps");
        window.location.reload();
      } else {
        render();
      }
    }, 1000);
  }

  function startPolling() {
    // 功能已關閉，節省讀取量。需由管理員手動發布總表。
    window.clearInterval(app.pollTimer);
  }

  function publicTemplate() {
    const term = app.search.trim();
    const filtered = app.students.filter(s => !term || (s.name || "").includes(term) || (s.classGroup || "").includes(term));
    const groups = {};
    filtered.forEach(student => {
      const group = student.classGroup || (student.name ? student.name.substring(0, 3) : "未分班");
      (groups[group] || (groups[group] = [])).push(student);
    });
    const classes = Object.keys(groups).sort();
    classes.forEach(group => {
      groups[group].sort((a, b) => (parseInt(a.seatNum || (a.name || "").slice(-2), 10) || 0) - (parseInt(b.seatNum || (b.name || "").slice(-2), 10) || 0));
    });

    return '<div class="container">' +
      '<div class="toolbar"><label class="searchbox">' + icons.search + '<input id="searchInput" type="text" placeholder="搜尋班級或學生..." value="' + escapeHtml(app.search) + '"></label></div>' +
      (classes.length ? classes.map(group => classTemplate(group, groups[group])).join("") : '<div class="empty">找不到符合條件的學生資料。</div>') +
      '</div>';
  }

  function classTemplate(group, students) {
    const topIds = new Set(students.filter(s => (s.views || 0) > 0).slice().sort((a, b) => ((b.views || 0) - (a.views || 0)) || ((b.likes || 0) - (a.likes || 0))).slice(0, 6).map(s => s.id));
    return '<section><div class="class-head"><h2 class="class-title">' + escapeHtml(group) + ' 班</h2><span class="pill">共 ' + students.length + ' 位學生</span></div>' +
      '<div class="grid">' + students.map(student => studentCardTemplate(student, topIds.has(student.id))).join("") + '</div></section>';
  }

      (isTop ? '<span class="tag tag-gold">' + icons.crown + '人氣王</span>' : "") +

  function normalizeUrl(url) {
    if (!url) return "#";
    return /^https?:\/\//i.test(url) ? url : "https://" + url;
  }

    if (event.target.id === "adminPassword") app.adminPassword = event.target.value;

    if (event.target.id === "studentUrl") app.newStudent.url = event.target.value;

    const formAction = event.target.getAttribute("data-action");
    if (!formAction) return;
    event.preventDefault();
    try {
      if (formAction === "login") {

        setMessage("success", "登入成功。");

    } catch (error) {
      console.error(error);
      setMessage("error", "操作失敗：" + error.message);
    }
  });

  root.addEventListener("click", async event => {
    const target = event.target.closest("[data-action]");
    if (!target) return;
    const action = target.getAttribute("data-action");
    const id = target.getAttribute("data-id");
    try {
      if (action === "dismiss-message") { app.message = null; render(); }
      if (action === "view-public") { app.view = "public"; render(); }
      if (action === "view-admin") { app.view = "admin"; render(); }
      if (action === "logout") { await signInAnonymous(); app.view = "admin"; setMessage("success", "已登出。"); }
      if (action === "view-work") await markOnce(id, "view", () => incrementStudent(id, "views"));
      if (action === "like") await markOnce(id, "like", () => incrementStudent(id, "likes"));
      if (action === "question") await markOnce(id, "question", () => incrementStudent(id, "questions"));
      if (action === "toggle-pick") {
        const student = app.students.find(s => s.id === id);
        await patchStudent(id, { isTeacherPick: !(student && student.isTeacherPick) });
      }
      if (action === "delete-student") await deleteStudent(id);
      if (action === "pick-csv") document.getElementById("csvInput").click();
      if (action === "reset-confirm") { app.modal = "reset"; render(); }
      if (action === "clear-confirm") { app.modal = "clear"; render(); }
      if (action === "close-modal") { app.modal = ""; render(); }
      if (action === "reset-all") await resetAll();
      if (action === "clear-all") await clearAll();
      if (action === "sync-directory") await syncDirectory();
    } catch (error) {
      console.error(error);
      setMessage("error", "操作失敗：" + error.message);
    }
  });

  root.addEventListener("change", event => {
    if (event.target.id === "csvInput") handleCsv(event.target.files[0], event.target);
  });

  async function markOnce(id, type, action) {
    const key = id + "_" + type;
    if (app.interactions[key]) return;
    app.interactions[key] = true;
    saveJson("student_interactions", app.interactions);
    render();
    await action();
  }

  async function handleCsv(file, input) {
    if (!file) return;
    const text = await file.text();
    const lines = text.split(/\r?\n/).filter(line => line.trim());
    let start = 0;
    if (lines[0] && (lines[0].includes("學生") || lines[0].toLowerCase().includes("name"))) start = 1;
    const students = [];
    for (let i = start; i < lines.length; i++) {
      const comma = lines[i].indexOf(",");
      if (comma <= 0) continue;
      const name = lines[i].slice(0, comma).trim().replace(/^"|"$/g, "");
      const url = lines[i].slice(comma + 1).trim().replace(/^"|"$/g, "");
      if (name) students.push({ name, classGroup: name.substring(0, 3) || "未分班", seatNum: name.slice(-2) || "00", url });
    }
    input.value = "";
    if (!students.length) return setMessage("error", "CSV 內沒有可匯入的學生資料。");
    const writes = students.map(student => ({ update: { name: firestoreBase().replace("https://firestore.googleapis.com/v1/", "") + "/students/" + crypto.randomUUID().replace(/-/g, ""), fields: writeFields(student).fields } }));
    await commitWrites(writes);
    setMessage("success", "已匯入 " + students.length + " 筆學生資料。");
  }

  async function resetAll() {
    const writes = app.students.map(student => ({ update: { name: documentName(student.id), fields: writeFields({ views: 0, likes: 0, questions: 0 }).fields }, updateMask: { fieldPaths: ["views", "likes", "questions"] } }));
    app.modal = "";
    await commitWrites(writes);
    setMessage("success", "已重置所有互動數據。");
  }

  async function clearAll() {
    const writes = app.students.map(student => ({ delete: documentName(student.id) }));
    app.modal = "";
    await commitWrites(writes);
    setMessage("success", "已清空全部學生資料。");
  }

  async function syncDirectory() {
    try {
      setMessage("success", "正在發布總表中，請稍候...");
      const data = await apiFetch(firestoreBase() + "/students?pageSize=500");
      const allStudents = (data.documents || []).map(docToStudent).filter(s => s.id !== "__directory__");
      
      await apiFetch(firestoreBase() + "/students/__directory__", {
        method: "PATCH",
        body: JSON.stringify({ fields: { data: { stringValue: JSON.stringify(allStudents) } } })
      });
      setMessage("success", "已將最新資料發布至前台總表！訪客的讀取量已降至 1 次。");
    } catch (e) {
      console.error(e);
      setMessage("error", "發布失敗：" + e.message);
    }
  }

  async function init() {
    if (checkRefreshLock()) return render();
    render();
    try {
      await signInAnonymous();
      await loadStudents();
    } catch (error) {
      console.error(error);
      setMessage("error", "資料庫連線失敗：" + error.message);
    }
  }

  init();
})();
